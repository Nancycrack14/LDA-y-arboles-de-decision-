from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.model_selection import StratifiedKFold, cross_val_score
import matplotlib.pyplot as plt
import numpy as np
from sklearn.metrics import accuracy_score, f1_score

# 1. Modelo inicial del rbol de decisin
tree = DecisionTreeClassifier(random_state=0)
tree.fit(X_train[variables_relevantes], y_train)

# Visualizar el rbol de decisin inicial
plt.figure(figsize=(15, 10))
plot_tree(tree, filled=True, feature_names=variables_relevantes, class_names=["No Victoria", "Victoria"], rounded=True)
plt.show()

# Realizar predicciones con el rbol de decisin inicial
yhat = tree.predict(X_test[variables_relevantes])

# Evaluar el rbol de decisin inicial
accuracy_initial = accuracy_score(y_test, yhat)
f1_initial = f1_score(y_test, yhat)
print(f"Accuracy inicial: {accuracy_initial}")
print(f"F1-score inicial: {f1_initial}")

# 2. Realizar validacin cruzada para encontrar el mejor valor de alpha (poda)
alpha_values = np.linspace(0.001, 0.2, 250)
cv_scores = []

# Realizar la validacin cruzada para cada valor de alpha
for alpha in alpha_values:
    tree_pruned = DecisionTreeClassifier(ccp_alpha=alpha, random_state=0)
    cv_score = np.mean(cross_val_score(tree_pruned, X_train[variables_relevantes], y_train, cv=StratifiedKFold(n_splits=5), scoring='f1'))
    cv_scores.append(cv_score)

# Encontrar el mejor valor de alpha
best_alpha = alpha_values[np.argmax(cv_scores)]
print(f"Mejor valor de alpha (): {best_alpha}")

# 3. rbol podado con el mejor alpha
pruned_tree = DecisionTreeClassifier(ccp_alpha=best_alpha, random_state=0)
pruned_tree.fit(X_train[variables_relevantes], y_train)

# Visualizar el rbol podado
plt.figure(figsize=(15, 10))
plot_tree(pruned_tree, filled=True, feature_names=variables_relevantes, class_names=["No Victoria", "Victoria"], rounded=True)
plt.show()

# Realizar predicciones con el rbol podado
yhat_pruned = pruned_tree.predict(X_test[variables_relevantes])

# Evaluar el rbol podado
accuracy_pruned = accuracy_score(y_test, yhat_pruned)
f1_pruned = f1_score(y_test, yhat_pruned)
print(f"Accuracy del rbol podado: {accuracy_pruned}")
